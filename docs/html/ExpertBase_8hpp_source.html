36 std::string parse_expert_signal(const std::string& raw_response, const std::string& expert_id) {
37 std::string signal;
38
39 // Padrões de parsing para cada especialista
40 std::regex pattern;
41
42 if (expert_id == "emotionalModel") {
43 pattern = std::regex(R"(\‍[SINAL\‍]\s*(\w+)\s*\‍[CONFIANÇA\‍]\s*(\d+\.?\d*)\s*\‍[JUSTIFICATIVA\‍]\s*(.+))");
44 } else if (expert_id == "analyticalModel") {
45 pattern = std::regex(R"(\‍[SINAL\‍]\s*(\w+)\s*\‍[CONFIANÇA\‍]\s*(\d+\.?\d*)\s*\‍[PADRÃO\‍]\s*(.+))");
46 } else if (expert_id == "memoryModel") {
47 pattern = std::regex(R"(\‍[FATO\‍]\s*(\w+)\s*\‍[CONFIANÇA\‍]\s*(\d+\.?\d*)\s*\‍[CONTEXTO\‍]\s*(.+))");
48 } else {
49 // Fallback para outros especialistas
50 pattern = std::regex(R"(\‍[SINAL\‍].+)");
51 }
52
53 std::smatch matches;
54 if (std::regex_search(raw_response, matches, pattern) && matches.size() >= 2) {
55 signal = matches[0];
56 } else {
57 // Se não seguir o formato, retorna sinal de erro
58 signal = "[ERRO] Formato inválido. Use: [SINAL] tipo [CONFIANÇA] 0.00-1.00 [CONTEXTO] info";
59 }
60
61 return signal;
62 }

140 std::string run(
141 const std::string& input,
142 alyssa_core::AlyssaCore* core_instance,
143 llama_adapter_lora* lora_override,
144 std::vector<llama_chat_message>& current_history,
145 llama_adapter_lora** active_lora_in_context,
146 std::function<void(const std::string&)> stream_callback = nullptr
147 ) override {
148 PCMetricsReader pcmetrics;
149
150 // 1. Adiciona mensagem do usuário ao histórico
151 std::string expert_input_with_role = "";
152 if (!config.role_instruction.empty()) {
153 expert_input_with_role = "[ROLE]: " + config.role_instruction + "\n" + input;
154 } else {
155 expert_input_with_role = input;
156 }
157
158 // IMPORTANTE: Verificar se precisamos alocar memória para a string
159 char* user_msg = strdup(expert_input_with_role.c_str());
160 current_history.push_back({"user", user_msg});
161
162 // 2. Monta template com métricas do sistema
163 std::vector<llama_chat_message> messages_to_template;
164 int system_prompt_index = -1;
165
166 std::string external = pcmetrics.get_simple_metrics_text();
167 std::string combined_system_prompt = config.system_prompt;
168
169 if (!external.empty()) {
170 combined_system_prompt += "\n[CONTEXTO DO SISTEMA - MÉTRICAS DO PC]:\n" + external;
171 }
172
173 if (!combined_system_prompt.empty()) {
174 system_prompt_index = messages_to_template.size();
175 char* sys_msg = strdup(combined_system_prompt.c_str());
176 messages_to_template.push_back({"system", sys_msg});
177 }
178
179 // Adiciona histórico da conversa
180 messages_to_template.insert(messages_to_template.end(),
181 current_history.begin(), current_history.end());
182
183 // 3. Aplica template
184 std::vector<char> formatted(core_instance->get_n_ctx() * 2); // Buffer maior
185 const char* tmpl = llama_model_chat_template(core_instance->get_model(), nullptr);
186
187 int len = llama_chat_apply_template(
188 tmpl, messages_to_template.data(), messages_to_template.size(),
189 true, formatted.data(), formatted.size()
190 );
191
192 // Limpa alocação do system prompt (se existir)
193 if (system_prompt_index != -1) {
194 free((char*)messages_to_template[system_prompt_index].content);
195 }
196
197 if (len < 0) {
198 // Tentar com buffer maior
199 formatted.resize(-len);
200 len = llama_chat_apply_template(
201 tmpl, messages_to_template.data(), messages_to_template.size(),
202 true, formatted.data(), formatted.size()
203 );
204 }
205
206 if (len < 0) {
207 return "Erro ao processar template de conversa.";
208 }
209
210 std::string prompt(formatted.begin(), formatted.begin() + len);
211
212 // 4. Executa geração
213 llama_adapter_lora* final_lora = (lora_override != nullptr) ? lora_override : lora;
214
215 // Atualizar o ponteiro ativo de LoRA
216 if (active_lora_in_context != nullptr) {
217 *active_lora_in_context = final_lora;
218 }
219
220 std::string response = core_instance->generate_raw(
221 prompt,
222 config.params,
223 final_lora,
224 stream_callback
225 );
226
227 // 5. Adiciona resposta ao histórico
228 char* assistant_msg = strdup(response.c_str());
229 current_history.push_back({"assistant", assistant_msg});
230
231 return response;
232 }

272 << ": " << e.what() << std::endl;
273 }
274 }
275
