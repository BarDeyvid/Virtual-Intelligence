17 std::string parse_expert_signal(const std::string& raw_response, const std::string& expert_id) {
18 std::string signal;
19
20 // Padrões de parsing para cada especialista
21 std::regex pattern;
22
23 if (expert_id == "emotionalModel") {
24 pattern = std::regex(R"(\‍[SINAL\‍]\s*(\w+)\s*\‍[CONFIANÇA\‍]\s*(\d+\.?\d*)\s*\‍[JUSTIFICATIVA\‍]\s*(.+))");
25 } else if (expert_id == "analyticalModel") {
26 pattern = std::regex(R"(\‍[SINAL\‍]\s*(\w+)\s*\‍[CONFIANÇA\‍]\s*(\d+\.?\d*)\s*\‍[PADRÃO\‍]\s*(.+))");
27 } else if (expert_id == "memoryModel") {
28 pattern = std::regex(R"(\‍[FATO\‍]\s*(\w+)\s*\‍[CONFIANÇA\‍]\s*(\d+\.?\d*)\s*\‍[CONTEXTO\‍]\s*(.+))");
29 } else {
30 // Fallback para outros especialistas
31 pattern = std::regex(R"(\‍[SINAL\‍].+)");
32 }
33
34 std::smatch matches;
35 if (std::regex_search(raw_response, matches, pattern) && matches.size() >= 2) {
36 signal = matches[0];
37 } else {
38 // Se não seguir o formato, retorna sinal de erro
39 signal = "[ERRO] Formato inválido. Use: [SINAL] tipo [CONFIANÇA] 0.00-1.00 [CONTEXTO] info";
40 }
41
42 return signal;
43 }

82 std::string run(
83 const std::string& input,
84 alyssa_core::AlyssaCore* core_instance,
85 llama_adapter_lora* lora_override,
86 std::vector<llama_chat_message>& current_history,
87 llama_adapter_lora** active_lora_in_context,
88 std::function<void(const std::string&)> stream_callback = nullptr
89 ) override {
90 PCMetricsReader pcmetrics;
91
92 // 1. Adiciona mensagem do usuário ao histórico
93 std::string expert_input_with_role = "";
94 if (!config.role_instruction.empty()) {
95 expert_input_with_role = "[ROLE]: " + config.role_instruction + "\n" + input;
96 } else {
97 expert_input_with_role = input;
98 }
99
100 // IMPORTANTE: Verificar se precisamos alocar memória para a string
101 char* user_msg = strdup(expert_input_with_role.c_str());
102 current_history.push_back({"user", user_msg});
103
104 // 2. Monta template com métricas do sistema
105 std::vector<llama_chat_message> messages_to_template;
106 int system_prompt_index = -1;
107
108 std::string external = pcmetrics.get_simple_metrics_text();
109 std::string combined_system_prompt = config.system_prompt;
110
111 if (!external.empty()) {
112 combined_system_prompt += "\n[CONTEXTO DO SISTEMA - MÉTRICAS DO PC]:\n" + external;
113 }
114
115 if (!combined_system_prompt.empty()) {
116 system_prompt_index = messages_to_template.size();
117 char* sys_msg = strdup(combined_system_prompt.c_str());
118 messages_to_template.push_back({"system", sys_msg});
119 }
120
121 // Adiciona histórico da conversa
122 messages_to_template.insert(messages_to_template.end(),
123 current_history.begin(), current_history.end());
124
125 // 3. Aplica template
126 std::vector<char> formatted(core_instance->get_n_ctx() * 2); // Buffer maior
127 const char* tmpl = llama_model_chat_template(core_instance->get_model(), nullptr);
128
129 int len = llama_chat_apply_template(
130 tmpl, messages_to_template.data(), messages_to_template.size(),
131 true, formatted.data(), formatted.size()
132 );
133
134 // Limpa alocação do system prompt (se existir)
135 if (system_prompt_index != -1) {
136 free((char*)messages_to_template[system_prompt_index].content);
137 }
138
139 if (len < 0) {
140 // Tentar com buffer maior
141 formatted.resize(-len);
142 len = llama_chat_apply_template(
143 tmpl, messages_to_template.data(), messages_to_template.size(),
144 true, formatted.data(), formatted.size()
145 );
146 }
147
148 if (len < 0) {
149 return "Erro ao processar template de conversa.";
150 }
151
152 std::string prompt(formatted.begin(), formatted.begin() + len);
153
154 // 4. Executa geração
155 llama_adapter_lora* final_lora = (lora_override != nullptr) ? lora_override : lora;
156
157 // Atualizar o ponteiro ativo de LoRA
158 if (active_lora_in_context != nullptr) {
159 *active_lora_in_context = final_lora;
160 }
161
162 std::string response = core_instance->generate_raw(
163 prompt,
164 config.params,
165 final_lora,
166 stream_callback
167 );
168
169 // 5. Adiciona resposta ao histórico
170 char* assistant_msg = strdup(response.c_str());
171 current_history.push_back({"assistant", assistant_msg});
172
173 return response;
174 }
