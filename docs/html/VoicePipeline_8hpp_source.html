97 class ThreadSafeQueue {
98 public:
103 void push(T value) {
104 std::lock_guard<std::mutex> lock(mtx);
105 q.push(std::move(value));
106 cv.notify_one();
107 }
108
115 bool pop(T& value) {
116 std::unique_lock<std::mutex> lock(mtx);
117 cv.wait(lock, [this] { return !running || !q.empty(); });
118
119 if (!running && q.empty()) {
120 return false;
121 }
122
123 value = std::move(q.front());
124 q.pop();
125 return true;
126 }
127
134 bool try_pop(T& value) {
135 std::lock_guard<std::mutex> lock(mtx);
136 if (q.empty()) {
137 return false;
138 }
139 value = std::move(q.front());
140 q.pop();
141 return true;
142 }
143
147 void stop() {
148 std::lock_guard<std::mutex> lock(mtx);
149 running = false;
150 cv.notify_all(); // Acorda todas as threads esperando em pop()
151 }
152 private:
153 std::queue<T> q;
154 std::mutex mtx;
155 std::condition_variable cv;

157 };
